# --- 6. Build the Hybrid Model ---

# Branch 1: The ANN for Tabular Data (Experiment 5)
input_tabular = Input(shape=(X_tabular.shape[1],), name='Tabular_Input')
ann_branch = Dense(32, activation='relu')(input_tabular)
ann_branch = Dense(16, activation='relu')(ann_branch)

# Branch 2: The DL/NLP Model for Text Data (Experiment 6)
input_text = Input(shape=(max_text_length,), name='Text_Input')
nlp_branch = Embedding(input_dim=vocab_size, output_dim=32, name='Embedding')(input_text)
nlp_branch = LSTM(32, name='LSTM_Layer')(nlp_branch) # LSTM is a Deep Learning layer

# Combine the two branches
combined = concatenate([ann_branch, nlp_branch], name='Concatenate_Branches')

# Add the final classifier head
final_classifier = Dense(16, activation='relu')(combined)
output = Dense(1, activation='sigmoid', name='Output_Layer')(final_classifier)

# Create the final multi-input model
hybrid_model = Model(inputs=[input_tabular, input_text], outputs=output)
